276: function getAttStats(students, attendance, classFilter, sectionFilter, date) {
277:   const classes = classFilter === "all" ? CLASSES : CLASSES.filter(c => c.id === classFilter);
278:   const sections = sectionFilter === "all" ? SECTIONS : [sectionFilter];
279:   let present = 0, total = 0;
280:   classes.forEach(cls => sections.forEach(sec => {
281:     const studs = students[`${cls.id}_${sec}`] || [];
282:     total += studs.length;
283:     const rec = attendance[`${date}_${cls.id}_${sec}`];
284:     if (rec) present += rec.present.length;
285:   }));
286:   return { present, total, absent: total - present, pct: total > 0 ? Math.round(present / total * 100) : 0 };
287: }
